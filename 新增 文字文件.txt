  if(ext==='xlsx'||ext==='xls'){processMomoExcel(f);return;}
  const r=new FileReader();
  r.onload=ev=>{
    const raw=ev.target.result.replace(/^\uFEFF/,'');const lines=raw.split('\n').filter(l=>l.trim());
    const isSh=lines.slice(0,5).some(l=>l.includes('Shopee台灣')||l.includes('所有CPC成效報告'));
    const hdr=parseCsvLine(lines[0]||'');
    const isCr=hdr.some(h=>h.toLowerCase().includes('campaign'))&&hdr.some(h=>h.toLowerCase().includes('displays'));
    if(isSh){processShopeeCsvContent(raw,lines);return;}
    if(isCr){processCriteoContent(raw,lines);return;}
    setStatus('importStatus','無法辨識格式，請確認是 Criteo CSV 或蝦皮 CSV','err');
  };r.readAsText(f,'UTF-8');
}
